          && rule.getAlertRouting().getDispatchLabelsList().contains(label.getKey())) {
        summaryLabelSubStrings.add(String.format("%s: %s", label.getKey(), label.getValue()));
      } else if (rule.hasAlertMatching()) {
        rule.getAlertMatching().getLabelsList().stream()
            .filter(l -> l.getKey().equals(label.getKey()))
            .map(NotificationLabel::getValue)
            .findFirst()
            .isPresent(
                value ->
                    summaryLabelSubStrings.add(String.format("%s: [%s]", label.getKey(), value)));
      }
    }
    summaryBuilder.append(String.format(" [%s]", String.join(", ", summaryLabelSubStrings)));
